test <- readr::read_csv("stats_countries_barcelona_2013-2023.csv")
## Rows: 185 Columns: 13
## ── Column specification ─────────────────────────────────────────────────────────────────────────────────────
## Delimiter: ","
## chr  (2): Country, Continent
## dbl (11): 2013, 2014, 2015, 2016, 2017, 2018, 2019, 2020, 2021, 2022, 2023
## 
## ℹ Use `spec()` to retrieve the full column specification for this data.
## ℹ Specify the column types or set `show_col_types = FALSE` to quiet this message.
# long format
test_long <- tidyr::pivot_longer(test, cols=-c(Country, Continent), names_to="Year", values_to="Count")
test_sub <- test_long %>% filter(Country %in% c("Francia", "Italia", "Alemania"))
ggplot(test_sub, aes(x=Year, y=Count, color=Continent)) + geom_line()

ggplot(test_sub, aes(x=Year, y=Count, fill=Country)) + geom_bar(stat="identity")

ggplot(test_sub, aes(x=Year, y=Count, fill=Country)) + geom_bar(stat="identity", position="dodge")
